📋 Guía de Configuración
1
Preparar el Spreadsheet

En tu Google Sheets creá dos hojas con estos nombres:

Clientes Movimientos

Hoja Clientes — fila 1 con headers:

id | nombre | telefono | direccion | email | diasCredito | estado | notas

Hoja Movimientos — fila 1 con headers:

id | clientId | tipo | fecha | monto | desc | notas
2
Abrir Apps Script

En tu Spreadsheet: Extensiones → Apps Script
Borrá todo y pegá este código:

function doGet(e) { var action = e.parameter.action; if (action === 'ping') { return resp({ok: true}); } if (action === 'read') { var ss = SpreadsheetApp.getActiveSpreadsheet(); var c = ss.getSheetByName('Clientes').getDataRange().getValues().slice(1); var m = ss.getSheetByName('Movimientos').getDataRange().getValues().slice(1); return resp({ok: true, clients: c, movements: m}); } return resp({ok: false}); } function doPost(e) { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(data.sheet); var rows = sheet.getDataRange().getValues(); var rowId = data.row[0]; for (var i = 1; i < rows.length; i++) { if (rows[i][0] == rowId) { sheet.getRange(i+1,1,1,data.row.length).setValues([data.row]); return resp({ok: true, action: 'updated'}); } } sheet.appendRow(data.row); return resp({ok: true, action: 'inserted'}); } function resp(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
